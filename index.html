                다시 시작

            </button>

        <!-- 5x4 그리드 (모바일에서는 카드가 작아짐) -->

        <div id="game-board" class="grid grid-cols-4 md:grid-cols-5 gap-2 md:gap-4 w-full max-w-4xl mx-auto">

            <!-- 자바스크립트로 카드가 생성되어 여기에 들어갑니다 -->

    <!-- 승리 모달창 -->

            </button>

    <script>

        /**

         * [중요] 이미지 파일 설정

         * 깃허브에 올릴 때, 10장의 사진 파일 이름을 아래 배열과 똑같이 맞춰서

         * 이 HTML 파일과 같은 폴더에 업로드하세요. (예: 1.jpeg, 2.jpeg ...)

         * 해상도는 카카오톡으로 전송해서 용량을 줄이거나 500x500 픽셀 정도로 자르는 것을 추천합니다.

         */

        const imageFiles = [

            '1.jpeg', '2.jpeg', '3.jpeg', '4.jpeg', '5.jpeg',

            '6.jpeg', '7.jpeg', '8.jpeg', '9.jpeg', '10.jpeg'

        ];


        const gameBoard = document.getElementById('game-board');

        const movesDisplay = document.getElementById('moves');

        const winModal = document.getElementById('win-modal');

        const finalMovesDisplay = document.getElementById('final-moves');

        

        let cards = [];

        let hasFlippedCard = false;

        let lockBoard = false;

        let firstCard, secondCard;

        let moves = 0;

        let matchedPairs = 0;

        const totalPairs = imageFiles.length;


        // 게임 초기화 함수

        function initGame() {

            gameBoard.innerHTML = '';

            moves = 0;

            matchedPairs = 0;

            movesDisplay.innerText = moves;

            winModal.classList.add('hidden');

            winModal.children[0].classList.replace('scale-100', 'scale-95');


            // 카드 2장씩 쌍으로 만들기

            let deck = [...imageFiles, ...imageFiles];

            // 카드 섞기 (Fisher-Yates 알고리즘)

            deck = deck.sort(() => Math.random() - 0.5);


            // HTML 카드 요소 생성

            deck.forEach((imagePath, index) => {

                const card = document.createElement('div');

                // 반응형 카드 크기 설정 (비율 유지)

                card.classList.add('card', 'perspective-1000', 'cursor-pointer', 'relative', 'aspect-[3/4]', 'w-full');

                card.dataset.image = imagePath;


                card.innerHTML = `

                    <div class="card-inner w-full h-full relative transform-style-3d rounded-xl shadow-md">

                        <!-- 카드 뒷면 (클릭 전 보이는 면) -->

                        <div class="card-back absolute w-full h-full backface-hidden bg-gradient-to-br from-indigo-500 to-purple-600 rounded-xl flex items-center justify-center border-4 border-white">

                            <span class="text-white text-4xl opacity-50">?</span>

                        </div>

                        <!-- 카드 앞면 (사진) -->

                        <div class="card-front absolute w-full h-full backface-hidden rotate-y-180 rounded-xl bg-white border-4 border-white shadow-inner overflow-hidden" 

                             style="background-image: url('${imagePath}');">

                        </div>

                    </div>

                `;


                card.addEventListener('click', flipCard);

                gameBoard.appendChild(card);

            });

        }


        // 카드 뒤집기 함수

        function flipCard() {

            if (lockBoard) return;

            if (this === firstCard) return;


            this.classList.add('flipped');


            if (!hasFlippedCard) {

                // 첫 번째 클릭

                hasFlippedCard = true;

                firstCard = this;

                return;

            }


            // 두 번째 클릭

            secondCard = this;

            moves++;

            movesDisplay.innerText = moves;

            checkForMatch();

        }


        // 짝이 맞는지 확인

        function checkForMatch() {

            let isMatch = firstCard.dataset.image === secondCard.dataset.image;


            if (isMatch) {

                disableCards();

            } else {

                unflipCards();

            }

        }


        // 짝이 맞았을 때 처리

        function disableCards() {

            firstCard.removeEventListener('click', flipCard);

            secondCard.removeEventListener('click', flipCard);


            // 매칭된 효과 주기

            firstCard.classList.add('matched');

            secondCard.classList.add('matched');


            matchedPairs++;

            

            if (matchedPairs === totalPairs) {

                setTimeout(gameWin, 500);

            }


            resetBoard();

        }


        // 짝이 틀렸을 때 원래대로 뒤집기

        function unflipCards() {

            lockBoard = true;


            setTimeout(() => {

                firstCard.classList.remove('flipped');

                secondCard.classList.remove('flipped');

                resetBoard();

            }, 1000); // 1초 뒤에 뒤집힘

        }


        // 상태 초기화

        function resetBoard() {

            [hasFlippedCard, lockBoard] = [false, false];

            [firstCard, secondCard] = [null, null];

        }


        // 게임 승리 처리

        function gameWin() {

            finalMovesDisplay.innerText = moves;

            winModal.classList.remove('hidden');

            setTimeout(() => {

                winModal.children[0].classList.replace('scale-95', 'scale-100');

            }, 10);

            

            // 축하 폭죽 효과

            var duration = 3 * 1000;

            var animationEnd = Date.now() + duration;

            var defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 100 };


            function randomInRange(min, max) {

              return Math.random() * (max - min) + min;

            }


            var interval = setInterval(function() {

              var timeLeft = animationEnd - Date.now();


              if (timeLeft <= 0) {

                return clearInterval(interval);

              }


              var particleCount = 50 * (timeLeft / duration);

              confetti(Object.assign({}, defaults, { particleCount,

                origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 }

              }));

              confetti(Object.assign({}, defaults, { particleCount,

                origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 }

              }));

            }, 250);

        }


        // 버튼 이벤트 리스너

        document.getElementById('restart-btn').addEventListener('click', initGame);

        document.getElementById('play-again-btn').addEventListener('click', initGame);


        // 게임 최초 실행

        initGame();

    </script>
